\section{TAD \tadNombre{Valor}}

\begin{tad}{\tadNombre{Valor}}
\tadIgualdadObservacional{v}{v'}{valor}{%
    \IF vEsString?($v$) THEN%
        vEsString?($v'$) $\yluego$ leerString($v$) \igobs leerString($v'$)%
    ELSE%
        vEsNat?($v'$) $\yluego$ leerNat($v$) \igobs leerNat($v'$)%
    FI%
}


\tadGeneros{valor}
\tadExporta{valor, generadores, observadores, otras operaciones}
\tadUsa{string, nat, bool}

\tadObservadores
\tadOperacion{leerString}{valor\ $v$}{string}{vEsString?($v$)}
\tadOperacion{leerNat}{valor\ $v$}{nat}{vEsNat?($v$)}
\tadOperacion{vEsString?}{valor}{bool}{}

\tadGeneradores
\tadOperacion{valorString}{string}{valor}{}
\tadOperacion{valorNat}{nat}{valor}{}

\tadOtrasOperaciones
\tadOperacion{vEsNat?}{valor}{bool}{}

\tadAxiomas[\paratodo{valor}{v}]
\tadAxioma{vEsString?(valorString(s))}{true}
\tadAxioma{vEsString?(valorNat(n))}{false}
\tadAxioma{vEsNat?(v)}{$\neg$vEsString?(v)}
\tadAxioma{leerString(valorString(s))}{s}
\tadAxioma{leerNat(valorNat(n))}{n}


\end{tad}
