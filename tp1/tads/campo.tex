\section{TAD \tadNombre{Campo}}

\begin{tad}{\tadNombre{Campo}}
    \tadIgualdadObservacional{c}{c'}{campo}{cNombre($c$) $\igobs$ cNombre($c'$) %
    $\land$ cEsString?($c$) $\igobs$ cEsString?($c'$)}

\tadGeneros{campo}
\tadExporta{campo, generadores, observadores, otras operaciones}
\tadUsa{string, bool}

\tadObservadores
\tadOperacion{cNombre}{campo}{string}{}
\tadOperacion{cEsString?}{campo}{bool}{}

\tadGeneradores
\tadOperacion{campoString}{string}{campo}{}
\tadOperacion{campoNat}{string}{campo}{}

\tadOtrasOperaciones
\tadOperacion{cEsNat?}{campo}{bool}{}

\tadAxiomas[\paratodo{campo}{c}]
\tadAxioma{cNombre(campoString(s))}{s}
\tadAxioma{cNombre(campoNat(s))}{s}
\tadAxioma{cEsString?(campoString(s))}{true}
\tadAxioma{cEsString?(campoNat(s))}{false}
\tadAxioma{cEsNat?(c)}{$\neg$ cEsString?(c)}

\end{tad}
